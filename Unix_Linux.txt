---------------------------------------------------------------------

or 
ssh-copy-id -i ~/.ssh/id_rsa.pub hadoop@slave1

chmod 700 ~/.ssh/authorized_keys
---------------------------------------------------------------------
-- Password less SSH FID1 to FID2 -----------------------------------
---------------------------------------------------------------------
hadoop1@hdpmc1$ 
	ssh-keygen -t rsa
	cksum ~/.ssh/id_rsa.pub
hadoop2@hdpmc2$
	mkdir ~/.ssh
	vi ~/.ssh/authorized_keys
	cksum ~/.ssh/authorized_keys
	# cksum should match 
	chmod 700 ~/.ssh
	chmod 400 ~/.ssh/authorized_keys
# Test
